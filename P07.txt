@echo off
setlocal

:: Folder containing your NDJSON files
set dir=C:\Users\Michael\Documents\json_files

:: Concatenate all files and feed them line by line into jq
type "%dir%\*.json" | jq -R -s '
  split("\n") 
  | map(select(length>0) | fromjson)
' > merged.json

echo Merged JSON written to merged.json
pause
