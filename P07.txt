import time
import psutil
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler

class NewFileHandler(FileSystemEventHandler):
    def on_created(self, event):
        if not event.is_directory:
            print(f"New file detected: {event.src_path}")

def print_resource_usage(process):
    cpu = psutil.cpu_percent(interval=None)
    mem = psutil.virtual_memory().percent
    proc_cpu = process.cpu_percent(interval=None)
    proc_mem = process.memory_info().rss / (1024 * 1024)

    print(f"[SYSTEM] CPU: {cpu}% | RAM: {mem}%")
    print(f"[PROCESS] CPU: {proc_cpu}% | RAM: {proc_mem:.2f} MB")

folder_to_watch = "/path/to/folder"

event_handler = NewFileHandler()
observer = Observer()
observer.schedule(event_handler, folder_to_watch, recursive=False)
observer.start()

process = psutil.Process()  # current Python process

try:
    while True:
        print_resource_usage(process)
        time.sleep(2)
except KeyboardInterrupt:
    observer.stop()

observer.join()
