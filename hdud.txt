#!/usr/bin/env python
import RPi.GPIO as GPIO
import time

pins = {'pin_R':11, 'pin_G':12, 'pin_B':13}  # pins is a dict

GPIO.setmode(GPIO.BOARD)      # Numbers GPIOs by physical location

for i in pins:
    GPIO.setup(pins[i], GPIO.OUT)  # Set pins' mode as output
    GPIO.output(pins[i], GPIO.HIGH)  # Set pins to high(+3.3V) to turn off led

p_R = GPIO.PWM(pins['pin_R'], 2000)  # set Frequency to 2KHz
p_G = GPIO.PWM(pins['pin_G'], 2000)
p_B = GPIO.PWM(pins['pin_B'], 2000)

p_R.start(0)  # Initial duty Cycle = 0 (leds off)
p_G.start(0)
p_B.start(0)

def set_colour(r_val, g_val, b_val):
    p_R.ChangeDutyCycle(r_val)
    p_G.ChangeDutyCycle(g_val)
    p_B.ChangeDutyCycle(b_val)

try:
    while True:
        print('Set Red')
        set_colour(0, 100, 100)
        time.sleep(1)

        print('Set Green')
        set_colour(100, 0, 100)
        time.sleep(1)

        print('Set Blue')
        set_colour(100, 100, 0)
        time.sleep(1)

        print('Set Yellow')
        set_colour(0, 0, 100)
        time.sleep(1)

        print('Set White')
        set_colour(0, 0, 0)
        time.sleep(1)

        print('Set Purple')
        set_colour(50, 100, 50)
        time.sleep(1)

        print('Set Off')
        set_colour(100, 100, 100)
        time.sleep(1)

except KeyboardInterrupt:
    p_R.start(0)
    p_G.start(0)
    p_B.start(0)
    for i in pins:
        GPIO.output(pins[i], GPIO.HIGH)  # Turn off all LEDs
    GPIO.cleanup()
